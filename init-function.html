function initPopOver(){ if($('[data-toggle=popover]').length > 0){ $('[data-toggle="popover"]').popover(); $('body').on('click', function (e) { $('[data-toggle=popover]').each(function () { // hide any open popovers when the anywhere else in the body is clicked if (!$(this).is(e.target) && $(this).has(e.target).length === 0 && $('.popover').has(e.target).length === 0) { $(this).popover('hide'); } }); }); } }; function initTooltip(){ if($('[data-toggle="tooltip"]').length > 0){ $('[data-toggle="tooltip"]').tooltip(); } };
